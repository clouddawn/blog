# 284_使用js判断是否为偶数的n种方法

## 一、%（取余）

```js
function isEven(num) {
  // 检查是否为数字类型且不是 NaN
  if (typeof num !== 'number' || Number.isNaN(num)) {
    return false;
  }
  
  // 检查是否为整数（包括负整数和零）
  if (!Number.isInteger(num)) {
    return false;
  }
  
  // 使用取余运算符判断是否为偶数
  return num % 2 === 0;
}
```

## 二、&（按位与）

```js
function isEven(num) {
  // 1. 检查是否为数字类型
  if (typeof num !== 'number' || isNaN(num)) {
    return false;
  }
  
  // 2. 检查是否为整数
  if (!Number.isInteger(num)) {
    return false;
  }
  
  // 3. 使用位运算判断奇偶性
  return (num & 1) === 0;
}
```

## 三、检查除以2后是否为整数

```js
function isEven(num) {
    // 1. 检查是否为数字类型
    if (typeof num !== 'number' || isNaN(num)) return false;
    
    // 2. 检查是否为整数
    if (!Number.isInteger(num)) return false;
    
    // 3. 检查除以2后是否为整数
    return Number.isInteger(num / 2);
}
```
